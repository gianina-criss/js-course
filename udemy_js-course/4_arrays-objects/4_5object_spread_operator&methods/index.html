   // spread operator
   const obj1 = { a: 1, b: 2 };
   const obj2 = { c: 5, d: 8 };
   const obj3 = { ...obj1, ...obj2 };
   x = obj3;
   console.log(x);
   
   //assign method
   const obj4 = Object.assign({}, obj1, obj2); // same thing like above
   x = obj4;
   console.log(x);
   
   //create an object in an array
   const todos = [
